Docker

官方文档：https://docs.docker.com/

Docker安装

# 首先卸载旧的版本
sudo yum remove docker \
                  docker-client \
                  docker-client-latest \
                  docker-common \
                  docker-latest \
                  docker-latest-logrotate \
                  docker-logrotate \
                  docker-engine
                  
# 安装环境
yum install -y yum-utils

# 设置镜像仓库
yum-config-manager --add-repo http://mirrors.aliyun.com/docker-ce/linux/centos/docker-ce.repo

# 安装最新版
yum install docker-ce docker-ce-cli containerd.io

# 启动docker
systemctl start docker

# 查看docker的信息
docker version

# 运行hello world进行测试
docker run hello-world

# 查看镜像
docker images

阿里云镜像加速

https://cr.console.aliyun.com/cn-qingdao/instances/mirrors

Docker常用命令

官方地址：https://docs.docker.com/reference/

帮助命令

# 基础信息
docker version
docker info
# 帮助命令
docker 命令 --help

镜像命令

镜像查找地址：https://hub.docker.com/

docker images []
-a # 列出所有的镜像
-q # 只显示镜像id
常用命令 docker images -aq # 显示所有镜像的id


# 查找镜像
docker search 镜像
docker search 镜像 -f=STARS=3000 查找star超出3000的镜像


# 下载镜像
docker pull 镜像


# 删除镜像
docker rmi -f 镜像id
docker rmi -r $(docker images -aq) # 删除所有的镜像

容器命令

docker run [可选参数] image

# 参数说明
--name="Name" # 给一个容器命名
-d 			  # 以后台方式运行
-it           # 使用交互方式运行，进入容器查看内容
-p            # 指定容器的端口 8080:8080 主机端口:容器端口

例子：
docker run -it centos /bin/bash # 


docker ps # 正在运行中的容器
docker ps -a # 历史运行过的容器

exit # 退出容器
Ctrl + P + Q # 容器不停止退出

docker rm # 容器id
docker rm -f # 强制删除

docker start 容器id # 启动停止的容器
docker restart 容器id # 重启容器
docker stop 容器id # 停止容器
docker kill 容器id

其他命令

docker logs
-t
-f
--tail 10

# 查看容器中的进程信息
docker top 容器id

# 查看容器元数据
docker inspect 容器id

进入正在运行的容器

# 进入容器id
docker exec -it 容器id bashshell # 打开一个新终端
docker attach 容器id # 进入容器正在执行的终端

从容器内拷贝到主机

docker ps 容器id:文件位置 主机目录

Docker卸载

yum remove docker-ce docker-ce-cli containerd.io

sudo rm -rf /var/lib/docker
